pipeline {
    agent any 
    stages {
        stage('Build docker images & install'){
            steps {
                 bat """ ECHO Get docker images and start them """
                 bat """ docker-compose up -d"""
            }
        }
        stage('Start testing'){
            steps {
                 bat """ ECHO Start the test scripts """
                 bat """ ECHO RUNNING TEST YADYADA """
            }
        }
        stage('Deploy'){
            steps {
                 bat """ ECHO Finished """
                 bat """ docker-compose down """
            }
        }
    }
}
